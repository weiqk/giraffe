422 void ShuntNetPacket::PreHandleADisconnection()
423 {
424  if(sock_delta_time_map_.empty())
425  {
426  return ;
427  }
428  else
429  {
430  map<zmq::socket_t *, unsigned long>::iterator it;
431  for(it=sock_delta_time_map_.begin();it!=sock_delta_time_map_.end();++it)
432  {
433  struct timeval tv;
434  gettimeofday(&tv,NULL);
435  if(0 == it->second)
436  {
437  it->second = tv.tv_sec*1000000 + tv.tv_usec;
438  }
439  else

451  for(iter_map=sock_ex_map_.begin();iter_map!=sock_ex_map_.end();)
